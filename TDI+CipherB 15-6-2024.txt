blueWaves_vwap(src, chlLen, avgLen) =>
    esa = ema(src, chlLen)
    d = ema(abs(src - esa), chlLen)
    ci = (src - esa) / (0.015 * d)
    bw1 = ema(ci, avgLen)
    bw2 = sma(bw1, tb)
    vwap = bw1 - bw2
    [bw1, bw2, vwap]
moneyFlowFunc(period, mult, y) => sma(((close - open) / (high - low)) * mult, period) - y
